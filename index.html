  <!-- 貼心一鍵複製代碼與輪播手勢腳本 -->

  <script>

    // --- 輪播邏輯機制 (僅在非 md 電腦版寬度下生效) ---

    let currentSlide = 0;

    const totalSlides = 2;


    function updateCarousel() {

      // 如果視窗寬度大於等於 768px (md)，不執行輪播位移

      if (window.innerWidth >= 768) {

        document.getElementById('carousel-track').style.transform = 'none';

        return;

      }


      const track = document.getElementById('carousel-track');

      track.style.transform = `translateX(-${currentSlide * 100}%)`;

      

      // 更新圓點樣式

      for (let i = 0; i < totalSlides; i++) {

        const dot = document.getElementById(`dot-${i}`);

        if (dot) {

          if (i === currentSlide) {

            dot.classList.remove('bg-white/30', 'hover:bg-white/50');

            dot.classList.add('bg-yellow-400', 'scale-110');

          } else {

            dot.classList.remove('bg-yellow-400', 'scale-110');

            dot.classList.add('bg-white/30', 'hover:bg-white/50');

          }

        }

      }

    }


    // 監聽視窗縮放，確保在桌機/手機切換時顯示正常

    window.addEventListener('resize', updateCarousel);


    function nextSlide() {

      if (window.innerWidth >= 768) return;

      currentSlide = (currentSlide + 1) % totalSlides;

      updateCarousel();

    }


    // --- 舊圖像、字卡的 onerror Fallback 機制 ---

    const allImages = document.querySelectorAll('img');

    allImages.forEach(img => {

      img.addEventListener('error', function() {

        if (this.alt === "注音海報") {

          this.src = 'https://images.unsplash.com/photo-1544377193-33dcf4d68fb5?auto=format&fit=crop&w=300&q=80';

        }

      });

    });


    function prevSlide() {

      if (window.innerWidth >= 768) return;

      currentSlide = (currentSlide - 1 + totalSlides) % totalSlides;

      updateCarousel();

    }


    function goToSlide(index) {

      if (window.innerWidth >= 768) return;

      currentSlide = index;

      updateCarousel();

    }


    // --- 手勢觸控滑動支援 (針對發燒網/嵌入框架深度優化) ---

    let startX = 0;

    let startY = 0;

    let isDragging = false;

    const track = document.getElementById('carousel-track');


    track.addEventListener('touchstart', (e) => {

      if (window.innerWidth >= 768) return;

      startX = e.touches[0].clientX;

      startY = e.touches[0].clientY;

      isDragging = true;

    }, { passive: true });


    track.addEventListener('touchend', (e) => {

      if (!isDragging || window.innerWidth >= 768) return;

      const endX = e.changedTouches[0].clientX;

      const endY = e.changedTouches[0].clientY;

      

      const diffX = startX - endX;

      const diffY = startY - endY;


      // 關鍵優化：只有當「左右滑動距離大於上下滑動距離」時，才執行輪播換頁

      if (Math.abs(diffX) > Math.abs(diffY)) {

        if (diffX > 50) {

          nextSlide();

        } else if (diffX < -50) {

          prevSlide();

        }

      }

      isDragging = false;

    }, { passive: true });


    // --- 一鍵複製折價券腳本 ---

    function copyCouponCode(code) {

      const tempTextArea = document.createElement('textarea');

      tempTextArea.value = code;

      tempTextArea.style.position = 'fixed'; 

      tempTextArea.style.opacity = '0';

      document.body.appendChild(tempTextArea);

      tempTextArea.select();

      

      try {

        const successful = document.execCommand('copy');

        if (successful) {

          const couponBtn = document.getElementById('coupon-btn');

          const originalHTML = couponBtn.innerHTML;

          

          couponBtn.innerHTML = `<i class="fa-solid fa-circle-check text-emerald-400 mr-1"></i> 已複製代碼！`;

          couponBtn.className = "inline-flex items-center justify-center text-xs sm:text-sm text-emerald-400 bg-emerald-500/10 border border-emerald-500/30 font-bold tracking-wide px-3.5 py-2 rounded-full shadow-sm transition-all w-fit mx-auto sm:mx-0";

          

          setTimeout(() => {

            couponBtn.innerHTML = originalHTML;

            couponBtn.className = "cursor-pointer group flex items-center justify-center gap-1.5 bg-white/10 border border-white/20 hover:bg-white/15 px-3.5 py-2 rounded-full text-xs sm:text-sm text-white font-bold active:scale-95 transition-all w-fit mx-auto sm:mx-0 shrink-0";

          }, 2000);

        }

      } catch (err) {

        console.error('複製失敗: ', err);

      } finally {

        document.body.removeChild(tempTextArea);

      }

    }

  </script>
